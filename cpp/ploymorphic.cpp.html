 2 #include <iostream>
 3 #include <string>
 4 
 5 class Shape {
 6 public:
 7   Shape(const std::string &name) : name(name) {}
 8 
 9   virtual double area() const = 0;
10   virtual void printInfo() const {
11     std::cout << "Shape: " << name << std::endl;
12   }
13 
14 private:
15   std::string name;
16 
17 protected:
18   const std::string &getName() const { return name; }
19 };
20 
21 class Circle : public Shape {
22 public:
23   Circle(const std::string &name, double radius)
24       : Shape(name), radius(radius) {}
25 
26   double area() const override { return 3.14159 * radius * radius; }
27 
28   void printInfo() const override {
29     std::cout << "Circle: " << Shape::getName() << ", Radius: " << radius
30               << ", Area: " << area() << std::endl;
31   }
32 
33 private:
34   double radius;
35 };
36 
37 class Rectangle : public Shape {
38 public:
39   Rectangle(const std::string &name, double length, double width)
40       : Shape(name), length(length), width(width) {}
41 
42   double area() const override { return length * width; }
43 
44   void printInfo() const override {
45     std::cout << "Rectangle: " << Shape::getName() << ", Length: " << length
46               << ", Width: " << width << ", Area: " << area() << std::endl;
47   }
48 
49 private:
50   double length;
51   double width;
52 };
53 
54 int main() {
55   Circle circle("Circle 1", 5.0);
56   Rectangle rectangle("Rectangle 1", 4.0, 6.0);
57 
58   Shape *shape1 = &circle;
59   Shape *shape2 = &rectangle;
60 
61   shape1->printInfo();
62   shape2->printInfo();
63 
64   return 0;
65 }
